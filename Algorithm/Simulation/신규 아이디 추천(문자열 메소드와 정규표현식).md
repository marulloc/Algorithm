# 신규 아이디 추천

```js
function solution(new_id) {
  let answer = new_id
    .toLowerCase()
    .replace(/[^\w-.]/g, "")
    .replace(/\.{2,}/g, ".")
    .replace(/^\.|\.$/, "");

  if (answer.length === 0) answer = "a";
  if (answer.length >= 16) answer = answer.slice(0, 15);
  answer = answer.replace(/\.$/, "");
  if (answer.length <= 2) {
    answer += answer[answer.length - 1].repeat(3 - answer.length);
  }

  return answer;
}
```
