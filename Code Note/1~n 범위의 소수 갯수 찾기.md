# 에라토스테네스의 체

```js
function eratosthenes(n) {
  // 1부터 n"이하"의 소수 개수 반환
  const isPrime = Array(n + 10).fill(true);

  for (let x = 2; x <= n; x++) {
    if (!isPrime[x]) continue;
    for (let mul = 2; mul <= n / x; mul++) {
      isPrime[x * mul] = false;
    }
  }

  let cntPrime = 0;
  for (let num = 2; num <= n; num++) {
    if (isPrime[num]) cntPrime++;
  }
  return cntPrime;
}
```
